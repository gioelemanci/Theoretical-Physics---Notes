\begin{definition}
  A \textbf{smooth function} on $\RR$ is a map $f:\RR \to \RR$ with continuous derivatives of all orders.
\end{definition}

\begin{theorem}[Fundamental example]
  If $f(x)$ is smooth and compactly supported, then its Fourier transform
  \[
    \hat{f}(k) = \frac{1}{\sqrt{2\pi}} \int_{-\infty}^\infty f(x) e^{-\ii kx} \dd x
  \]
  is smooth and rapidly decreasing.
\end{theorem}\TODO{rewrite.}

\begin{proof}
  Repeated integration by parts shows that every derivative $\hat{f}^{(n)}(k)$ is bounded by $\norm{x^n f(x)}_{L^1}$, hence smoothness and decay follow.
\end{proof}

\begin{corollary}
  If $f$ is smooth and compactly supported, then $\hat{f}$ belongs to the Schwartz space $\mathcal{S}(\RR)$.
\end{corollary}
